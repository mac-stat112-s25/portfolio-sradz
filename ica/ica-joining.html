contact <- data.frame(
  name = c("A", "B", "C", "D"),
  address = c("summit", "grand", "snelling", "fairview"),
  age = c(24, 89, 43, 38)
)

contact

# 1. We want contact info for people who HAVEN'T voted
contact |>
  anti_join(voters, join_by(name == id))
  name  address age

contact |>
  semi_join(voters, join_by(name == id))
  name  address age
